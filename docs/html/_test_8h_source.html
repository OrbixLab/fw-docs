297inline bool Test::position_validation(Piece _piece, uint8_t expected_tile)
298{
299 // Verifica que el índice esté dentro del rango válido
300 if (expected_tile >= MAX_TILES)
301 {
302 Serial.println("Error: expected_tile fuera de rango");
303 return false;
304 }
305
306 for (uint8_t i = 0; i < MAX_COL; i++)
307 {
308
309 if (tiles[expected_tile][i].positionX == NO_POSITION && tiles[expected_tile][i].positionY == NO_POSITION)
310 {
311 continue; // Ignora esta posición y pasa a la siguiente
312 }
313
314 if (tiles[expected_tile][i].positionX == _piece.positionX && tiles[expected_tile][i].positionY == _piece.positionY)
315 {
316 Serial.println("Position correct");
317 return true;
318 }
319 }
320 Serial.println("Position INcorrect");
